// Programa que eleva un numero x a la potencia y

main {
  // Declarar variables:
  int x,y;
  int i,total;

  // Leer x e y:
  scanf x;
  scanf y;

  // Calcular la potencia:
  i = 1;
  total = 1;
  while (i <= y) {
    total = total*x;
    i = i + 1;
  }

  // Mostrar resultado:
  printf total;
}
